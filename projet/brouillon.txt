<script>
 function ajouter() {
        var newQualitesInput = document.getElementById("newQualites");
        var checkboxes = document.getElementById("checkboxes");
        var newQualitesValue = newQualitesInput.value;

        if (newQualitesValue !== "") {
            var checkbox = document.createElement("input");
            checkbox.type = "checkbox";
            checkbox.name = "qualites[]";
            checkbox.value = newQualitesValue;

            var label = document.createElement("label");
            label.appendChild(checkbox);
            label.appendChild(document.createTextNode(newQualitesValue));

            var br = document.createElement("br");

            checkboxes.appendChild(label);
            checkboxes.appendChild(br);

            newQualitesInput.value = "";

            // Désactive le bouton "Ajouter" après trois ajouts
            if (checkboxes.getElementsByTagName("input").length > 13) {
                document.getElementById("ajouterBtn").disabled = true;
            }
        }
    }
</script>

            var afficher = document.getElementById("mdp");
            var eye = document.getElementById("eye");
            if(afficher.type === "password"){
